# Ecotrax MVP — Plan 2: Source Cache & ETL

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Populate every Layer 1 source-cache table (`src_*`) with real data from real sources (GBIF, IUCN Red List, WDPA, Global Forest Watch, LandMark), plus build the `species_id_map` cross-source reconciliation. Establish the ETL infrastructure (job tracking, scheduled GitHub Actions, Vercel Cron) that Plans 3+ rely on.

**Architecture:** Two runtime tiers. Heavy bulk pulls (IUCN bulk, WDPA shapefiles, range maps, LandMark, reconciliation) run as scheduled GitHub Actions invoking CLI scripts under `scripts/etl/*.ts`. Short-cycle work (GBIF on-demand species fetch, GFW daily alert delta) runs as Next.js API routes triggered by Vercel Cron. Both tiers share a single `lib/etl/` library that handles HTTP clients, geometry/H3 transforms, and database writes. Every job records its run in `materialization_jobs` and updates `source_refresh_log`.

**Tech Stack additions:** `h3-js@4` (H3 cell indexing in TS), `wellknown` (WKT round-trip), `shapefile` (npm package; streaming shapefile parser), `@supabase/supabase-js` (already installed; used directly with the secret key for ETL writes), no PostGIS-specific Node libs needed (we generate WKT strings and let Postgres parse them).

**Spec reference:** [`docs/superpowers/specs/2026-05-03-ecotrax-mvp-design.md`](../specs/2026-05-03-ecotrax-mvp-design.md), §5 (Data Integration & ETL).

**Plan 1 baseline (HEAD):** `060e7a9`. All Layer 1 schema is in place but every `src_*` table is empty; this plan fills them.

```ts
import { latLngToCell } from 'h3-js'
import { parse as parseWkt } from 'wellknown'

export interface H3Cells {
  r5: string
  r6: string
  r7: string
}

export function cellsForPoint(lat: number, lng: number): H3Cells {
  return {
    r5: latLngToCell(lat, lng, 5),
    r6: latLngToCell(lat, lng, 6),
    r7: latLngToCell(lat, lng, 7),
  }
}

export function cellForPolygonCenter(wkt: string, resolution: number): string {
  const parsed = parseWkt(wkt)
  if (!parsed) throw new Error(`unparseable WKT: ${wkt}`)
  const [lng, lat] = centroid(parsed)
  return latLngToCell(lat, lng, resolution)
}

// Simple unweighted centroid of all coords. Sufficient for assigning a
// single cell to a polygon at our resolution. NOT a true geometric centroid.
function centroid(g: GeoJSON.Geometry): [number, number] {
  const points = collectCoords(g)
  if (points.length === 0) throw new Error('empty geometry')
  const sum = points.reduce<[number, number]>(
    ([sx, sy], [x, y]) => [sx + x, sy + y],
    [0, 0],
  )
  return [sum[0] / points.length, sum[1] / points.length]
}

function collectCoords(g: GeoJSON.Geometry): [number, number][] {
  switch (g.type) {
    case 'Point': return [[g.coordinates[0], g.coordinates[1]]]
    case 'MultiPoint':
    case 'LineString': return g.coordinates.map((c) => [c[0], c[1]])
    case 'MultiLineString':
    case 'Polygon': return g.coordinates.flat().map((c) => [c[0], c[1]])
    case 'MultiPolygon': return g.coordinates.flat(2).map((c) => [c[0], c[1]])
    case 'GeometryCollection':
      return g.geometries.flatMap(collectCoords)
  }
}
```

`lib/etl/geometry.ts`:

```ts
export function isFiniteCoord(lat: number, lng: number): boolean {
  return Number.isFinite(lat) && Number.isFinite(lng)
    && lat >= -90 && lat <= 90
    && lng >= -180 && lng <= 180
}

export function pointWkt(lng: number, lat: number): string {
  if (!Number.isFinite(lng) || !Number.isFinite(lat)) {
    throw new Error(`pointWkt: non-finite coord ${lng},${lat}`)
  }
  return `POINT(${lng} ${lat})`
}

export function polygonWktFromGeoJson(
  geo: GeoJSON.Polygon | GeoJSON.MultiPolygon,
): string {
  if (geo.type === 'Polygon') {
    return `POLYGON(${ringsToWkt(geo.coordinates)})`
  }
  return `MULTIPOLYGON(${geo.coordinates.map((p) => `(${ringsToWkt(p)})`).join(',')})`
}

function ringsToWkt(rings: number[][][]): string {
  return rings.map((ring) =>
    `(${ring.map(([x, y]) => `${x} ${y}`).join(',')})`,
  ).join(',')
}
```

```ts
import type { SupabaseClient } from '@supabase/supabase-js'

export interface StartJobInput {
  jobType: string
  source?: string
  scope?: Record<string, unknown>
}

export async function startJob(
  client: SupabaseClient,
  input: StartJobInput,
): Promise<number> {
  const { data, error } = await client
    .from('materialization_jobs')
    .insert({
      job_type: input.jobType,
      source: input.source,
      scope: input.scope,
      started_at: new Date().toISOString(),
      status: 'running',
    })
    .select('id')
    .single()
  if (error) throw error
  return (data as { id: number }).id
}

export interface FinishJobInput {
  status: 'success' | 'failed'
  rowsProcessed?: number
  rowsFailed?: number
  errorLog?: string
}

export async function finishJob(
  client: SupabaseClient,
  jobId: number,
  input: FinishJobInput,
): Promise<void> {
  const { error } = await client
    .from('materialization_jobs')
    .update({
      finished_at: new Date().toISOString(),
      status: input.status,
      rows_processed: input.rowsProcessed ?? 0,
      rows_failed: input.rowsFailed ?? 0,
      error_log: input.errorLog ?? null,
    })
    .eq('id', jobId)
  if (error) throw error
}

export interface RecordRefreshInput {
  source: string
  recordsAdded?: number
  recordsUpdated?: number
  nextDueAt?: Date
}

export async function recordSourceRefresh(
  client: SupabaseClient,
  input: RecordRefreshInput,
): Promise<void> {
  const { error } = await client.from('source_refresh_log').insert({
    source: input.source,
    last_synced_at: new Date().toISOString(),
    next_due_at: input.nextDueAt?.toISOString(),
    records_added: input.recordsAdded ?? 0,
    records_updated: input.recordsUpdated ?? 0,
  })
  if (error) throw error
}
```

```ts
const BASE = 'https://api.gbif.org/v1'

export interface GbifTaxon {
  key: number
  scientificName: string
  vernacularName?: string
  rank?: string
  class?: string
  family?: string
  status?: string
}

export interface GbifOccurrence {
  key: number | string
  decimalLatitude?: number
  decimalLongitude?: number
  coordinateUncertaintyInMeters?: number
  eventDate?: string
  taxonKey: number
  scientificName?: string
  basisOfRecord?: string
  datasetKey?: string
  taxonomicStatus?: string
  identificationVerificationStatus?: string
  // ...many more fields preserved as `raw`
  [k: string]: unknown
}

export async function fetchTaxon(taxonKey: number): Promise<GbifTaxon> {
  const r = await fetch(`${BASE}/species/${taxonKey}`, {
    headers: { 'User-Agent': 'Ecotrax/0.1 (+https://github.com/muralisreedhar/ecotrax)' },
  })
  if (!r.ok) throw new Error(`GBIF /species/${taxonKey}: ${r.status}`)
  return await r.json() as GbifTaxon
}

export interface OccurrenceQuery {
  limit?: number          // page size; GBIF max 300
  hasCoordinate?: boolean
  hasGeospatialIssue?: boolean
}

export async function* fetchOccurrencesByTaxon(
  taxonKey: number,
  q: OccurrenceQuery = {},
): AsyncGenerator<GbifOccurrence[]> {
  const limit = Math.min(q.limit ?? 300, 300)
  let offset = 0
  while (true) {
    const params = new URLSearchParams({
      taxonKey: String(taxonKey),
      limit: String(limit),
      offset: String(offset),
      hasCoordinate: String(q.hasCoordinate ?? true),
      hasGeospatialIssue: String(q.hasGeospatialIssue ?? false),
    })
    const r = await fetch(`${BASE}/occurrence/search?${params}`, {
      headers: { 'User-Agent': 'Ecotrax/0.1' },
    })
    if (!r.ok) throw new Error(`GBIF /occurrence/search: ${r.status}`)
    const body = await r.json() as {
      offset: number
      limit: number
      endOfRecords: boolean
      results: GbifOccurrence[]
    }
    if (body.results.length > 0) yield body.results
    if (body.endOfRecords || body.results.length === 0) return
    offset += limit
  }
}
```

```ts
import type { SupabaseClient } from '@supabase/supabase-js'
import { fetchOccurrencesByTaxon } from './client'
import { cellsForPoint } from '@/lib/etl/h3'
import { isFiniteCoord, pointWkt } from '@/lib/etl/geometry'

export interface SyncResult {
  added: number
  updated: number
  skipped: number
}

export async function syncSpeciesOccurrences(
  client: SupabaseClient,
  gbifKey: number,
): Promise<SyncResult> {
  let added = 0, updated = 0, skipped = 0

  for await (const page of fetchOccurrencesByTaxon(gbifKey)) {
    const rows = page.flatMap((occ) => {
      const lat = occ.decimalLatitude
      const lng = occ.decimalLongitude
      if (typeof lat !== 'number' || typeof lng !== 'number' || !isFiniteCoord(lat, lng)) {
        skipped++
        return []
      }
      const cells = cellsForPoint(lat, lng)
      return [{
        gbif_key: gbifKey,
        occurrence_id: String(occ.key),
        location: pointWkt(lng, lat),
        h3_r5: cells.r5,
        h3_r6: cells.r6,
        h3_r7: cells.r7,
        observed_at: occ.eventDate ?? null,
        source: occ.datasetKey ?? null,
        coordinate_uncertainty_m: occ.coordinateUncertaintyInMeters ?? null,
```

```ts
/**
 * Walks every IUCN species page, then for each species fetches detail + threats
 * and upserts into src_iucn_assessments. Vertebrate-class filter applied
 * post-detail (page list doesn't include class).
 */
export async function importAllAssessments(
  client: SupabaseClient,
  options: { startPage?: number; endPage?: number; vertebratesOnly?: boolean } = {},
): Promise<ImportResult> {
  let pagesProcessed = 0
  let assessmentsUpserted = 0
  let errors = 0

  // IUCN paginates to ~16 pages of ~10000 species each as of 2026.
  const startPage = options.startPage ?? 0
  const endPage = options.endPage ?? 20

  for (let p = startPage; p < endPage; p++) {
    const list = await fetchSpeciesPage(p)
    if (list.length === 0) break
    pagesProcessed++

    for (const item of list) {
      try {
        const detail = await fetchSpeciesDetail(item.taxonid)
        if (!detail) continue
        const threats = await fetchSpeciesThreats(item.taxonid)
        const row = {
          iucn_id: detail.taxonid,
          scientific_name: detail.scientific_name,
          iucn_status: detail.category,
          population_trend: detail.population_trend ?? null,
          assessment_date: detail.assessment_date
            ? new Date(detail.assessment_date).toISOString()
            : null,
          threats_raw: threats,
          narratives_raw: {
            rationale: detail.rationale,
            habitat: detail.habitat,
            threats: detail.threats,
            conservation_measures: detail.conservationmeasures,
          },
          raw_payload: detail as Record<string, unknown>,
        }
        const { error } = await client
          .from('src_iucn_assessments')
          .upsert(row, { onConflict: 'iucn_id' })
        if (error) throw error
        assessmentsUpserted++
      } catch (e) {
        errors++
        console.warn(`iucn:${item.taxonid}: ${e instanceof Error ? e.message : e}`)
      }
    }
  }

  return { pagesProcessed, assessmentsUpserted, errors }
}
```

```ts
import type { SupabaseClient } from '@supabase/supabase-js'
import { open } from 'shapefile'
import { polygonWktFromGeoJson } from '@/lib/etl/geometry'

export interface ImportRangesResult { processed: number; failed: number }

export async function importRangeShapefile(
  client: SupabaseClient,
  shpPath: string,
): Promise<ImportRangesResult> {
  let processed = 0, failed = 0
  const source = await open(shpPath)
  // shapefile streams Feature<Geometry, Properties>
  while (true) {
    const next = await source.read()
    if (next.done) break
    const f = next.value
    if (!f) continue
    try {
      const props = (f.properties ?? {}) as Record<string, unknown>
      const iucnId = Number(props.id_no ?? props.taxonid ?? props.SISID)
      if (!Number.isInteger(iucnId)) { failed++; continue }
      const wkt = polygonWktFromGeoJson(f.geometry as GeoJSON.Polygon | GeoJSON.MultiPolygon)
      const { error } = await client.from('src_iucn_ranges').insert({
        iucn_id: iucnId,
        range_polygon: wkt,
        presence_code: numOrNull(props.presence),
        origin_code: numOrNull(props.origin),
        seasonal_code: numOrNull(props.seasonal),
      })
      if (error) throw error
      processed++
    } catch {
      failed++
    }
  }
  return { processed, failed }
}

function numOrNull(v: unknown): number | null {
  return typeof v === 'number' ? v : null
}
```

```ts
const DATASET = 'gfw_integrated_alerts/latest'

export interface GfwAlert {
  alert_date: string
  longitude: number
  latitude: number
  confidence: 'low' | 'high' | 'highest'
  alert__id: string
}

function key(): string {
  const k = process.env.GFW_API_KEY
  if (!k) throw new Error('GFW_API_KEY not set. Register at https://www.globalforestwatch.org/.')
  return k
}

export async function fetchRecentAlerts(sinceIso: string, limit = 10000): Promise<GfwAlert[]> {
  // The data-api supports SQL-style queries against the dataset
  const sql = `SELECT longitude, latitude, alert__date AS alert_date, confidence__cat AS confidence
               FROM results
               WHERE alert__date >= '${sinceIso.slice(0, 10)}'
               LIMIT ${limit}`
  const r = await fetch(`${BASE}/dataset/${DATASET}/query?sql=${encodeURIComponent(sql)}`, {
    headers: { 'x-api-key': key() },
  })
  if (!r.ok) throw new Error(`GFW query: ${r.status} ${await r.text()}`)
  const body = await r.json() as { data: Omit<GfwAlert, 'alert__id'>[] }
  return body.data.map((a, i) => ({ ...a, alert__id: `${a.alert_date}:${a.latitude}:${a.longitude}:${i}` }))
}
```

```ts
import { fetchRecentAlerts } from './client'
import { cellsForPoint } from '@/lib/etl/h3'
import { isFiniteCoord, pointWkt } from '@/lib/etl/geometry'

export async function syncRecentAlerts(
  client: SupabaseClient,
  sinceHours = 24,
): Promise<{ inserted: number; skipped: number }> {
  const since = new Date(Date.now() - sinceHours * 3600_000).toISOString()
  const alerts = await fetchRecentAlerts(since)

  let inserted = 0, skipped = 0
  const rows = alerts.flatMap((a) => {
    if (!isFiniteCoord(a.latitude, a.longitude)) { skipped++; return [] }
    return [{
      alert_type: 'integrated',
      location: pointWkt(a.longitude, a.latitude),
      h3_r6: cellsForPoint(a.latitude, a.longitude).r6,
      detected_at: a.alert_date,
      confidence_level: a.confidence,
      raw_payload: a as Record<string, unknown>,
    }]
  })

  if (rows.length > 0) {
    const { error, count } = await client
      .from('src_gfw_alerts')
      .insert(rows, { count: 'exact' })
    if (error) throw error
    inserted = count ?? rows.length
  }
  return { inserted, skipped }
}
```

```ts
  try {
    const result = await syncRecentAlerts(client, 24)
    await finishJob(client, jobId, {
      status: 'success', rowsProcessed: result.inserted, rowsFailed: result.skipped,
    })
    await recordSourceRefresh(client, {
      source: 'gfw', recordsAdded: result.inserted,
      nextDueAt: new Date(Date.now() + 24 * 3600_000),
    })
    return NextResponse.json({ ok: true, ...result })
  } catch (err) {
    await finishJob(client, jobId, {
      status: 'failed', errorLog: err instanceof Error ? err.message : String(err),
    })
    return NextResponse.json({ ok: false }, { status: 500 })
  }
}

export const GET = POST
```

```ts
import type { SupabaseClient } from '@supabase/supabase-js'

export interface MatchResult {
  method: 'exact' | 'fuzzy' | 'none'
  confidence: number
}

export function matchClassify(a: string, b: string): MatchResult {
  if (a === b) return { method: 'exact', confidence: 1 }
  const sim = trigramSimilarity(a.toLowerCase(), b.toLowerCase())
  if (sim >= 0.85) return { method: 'fuzzy', confidence: sim }
  return { method: 'none', confidence: sim }
}

// Lightweight trigram similarity. (Postgres's pg_trgm does the same on the
// query side; this lets us classify in JS for tests + filtering.)
function trigramSimilarity(a: string, b: string): number {
  const ta = trigrams(a), tb = trigrams(b)
  if (ta.size === 0 || tb.size === 0) return 0
  const inter = [...ta].filter((g) => tb.has(g)).length
  return inter / Math.max(ta.size, tb.size)
}
function trigrams(s: string): Set<string> {
  const padded = `  ${s} `
  const out = new Set<string>()
  for (let i = 0; i < padded.length - 2; i++) out.add(padded.slice(i, i + 3))
  return out
}

/**
 * For every src_iucn_assessments row, find the GBIF taxonKey via:
 *   1. exact match on scientific_name
 *   2. trigram similarity > 0.85 (uses pg_trgm)
 * Writes/updates species_id_map with method + confidence.
 *
 * Uses GBIF's species/match endpoint as the lookup; results are cached in
 * src_gbif_occurrences.gbif_key but we go to GBIF /species/match for
 * authoritative reconciliation.
 */
export async function reconcileSpeciesIdMap(
  client: SupabaseClient,
): Promise<{ matched: number; ambiguous: number }> {
  let matched = 0, ambiguous = 0

  // Pull every IUCN assessment we know about
  const { data: assessments, error } = await client
    .from('src_iucn_assessments')
    .select('iucn_id, scientific_name')
  if (error) throw error
  if (!assessments) return { matched, ambiguous }

  for (const row of assessments as { iucn_id: number; scientific_name: string }[]) {
    const gbifKey = await gbifMatchByName(row.scientific_name)
    if (!gbifKey) { ambiguous++; continue }

    await client.from('species_id_map').upsert({
      canonical_species_id: gbifKey.usageKey,
      source_system: 'iucn',
      source_id: String(row.iucn_id),
      match_method: gbifKey.matchType.toLowerCase() === 'exact' ? 'exact' : 'fuzzy',
      match_confidence: gbifKey.confidence ?? 0.85,
      manual_override: false,
    }, { onConflict: 'source_system,source_id' })
    matched++
  }
  return { matched, ambiguous }
}

interface GbifMatch {
  usageKey: number
  matchType: string  // 'EXACT' | 'FUZZY' | 'HIGHERRANK' | 'NONE'
  confidence?: number
}

async function gbifMatchByName(name: string): Promise<GbifMatch | null> {
  const r = await fetch(
    `https://api.gbif.org/v1/species/match?strict=false&verbose=false&name=${encodeURIComponent(name)}`,
  )
  if (!r.ok) return null
  const body = await r.json() as GbifMatch
  if (!body.usageKey || body.matchType === 'NONE') return null
  return body
}
```
